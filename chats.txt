def preprocess_text(text):
    text = contractions.fix(text)
    # Remove mentions after link or name detection
    text = re.sub(r'((http|https)\S+)|(www.\S+)|(@\S+)', '', text)
    
    # Remove non-alphabetic characters using regular expressions
    text = re.sub(r'[^a-zA-Z\']', ' ', text)

    # Convert the text to lowercase
    text = text.lower()
    text = re.sub(r'\b(i|today|girl|vaccine|covid|astrazeneca|pfizer|covidvaccine|russian|pharmaceutical|think|us|s)\b', '', text)
    
    # Tokenize the text using spaCy
    doc = nlp(text)

    # Apply stemming and remove stopwords
    processed_tokens = [token.lemma_ for token in doc if token.lemma_ not in STOP_WORDS]

    # Join the processed tokens back into a single string
    processed_text = ' '.join(processed_tokens)

    return processed_text
